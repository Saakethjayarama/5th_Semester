    def gcd(m, n):
        fm = []
        fn = []
        for i in range(1, m+1):
            if (m%i) == 0:
                fm.append(i)

        for i in range(1, n+1):
            if (n%i) == 0:
                fn.append(i)

        cm = []
        for j in fn:
            if j in fm:
                cm.append(j)

        return cm[-1]

class KeyNotFoundException(Exception):
    pass

def bs(lst, key):
    low = 0
    high = len(lst)
    while low < high:
        mid = (low + high) / 2
        if lst[mid] == key:
            return mid
        elif lst[mid] > key:
            high = mid - 1
        else:
            low = mid + 1
    raise KeyNotFoundException
