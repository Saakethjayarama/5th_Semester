def ascending(l):
    for i in range(len(l)):
        if (i == 0):
            continue
        else:
            if(l[i] < l[i - 1]):
                return False
    return True

def ascending(l):
    if len(l) < 2:
        return True
    return (l[0] <= l[1] and ascending(l[1:]))
